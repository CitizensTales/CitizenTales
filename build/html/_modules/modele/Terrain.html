[docs]    def translateBackLevel(self, matrice):
        return [[self.allowedtiles[x] for x in y] for y in matrice]



[docs]    def translateMidLevel(self, matrice):
        act = Acteur.allowedActors()
        actList = []
        x, y, z = 0, 0, 0
        for ligne in matrice:
            for case in ligne:
                if act[case] is not None:
                    actor = copy.deepcopy(act[case])
                    actor.deplacementAbsolu(x, y)
                    actList.append(actor)
                    if act[case].getNom() == "Hero":
                        self.indiceHero = z
                    z += 1
                x += 1
            x = 0
            y += 1
        return actList
